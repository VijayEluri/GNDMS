00078         if (Object.class.equals(clazz))
00079             return;
00080         else
00081             fillParamMapForClass(clazz.getSuperclass(), mapParam);
00082         for (Field field : clazz.getDeclaredFields()) {
00083             ConfigOption option = field.getAnnotation(ConfigOption.class);

00091                         mapParam.put(altName, option);
00092                 }
00093                 else
00094                     mapParam.put(field.getName(), option);

00096         }
